#include <stdio.h>
int main()
{
    int block[3]={100,200,300};
    int process[3]={80,150,250};
    int i,j;
    for(i=0;i<3;i++)
    {
        for(j=0;j<3;j++)
        {
            if(process[i]<=block[j])
            {
                printf("Process %d allocated to Block %d\n",i+1,j+1);
                block[j]=0;
                break;
            }
        }
    }
    return 0;
}
