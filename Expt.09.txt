#include <stdio.h>
#include <sys/ipc.h>
#include <sys/shm.h>
int main()
{
    int shmid;
    char *str;
    shmid = shmget(1234, 1024, 0666 | IPC_CREAT);
    str = (char *)shmat(shmid, NULL, 0);
    printf("Enter data: ");
    scanf("%s", str);
    printf("Data in shared memory: %s", str);
    shmdt(str);
    return 0;
}
